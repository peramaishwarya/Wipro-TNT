import java.io.*;
import java.util.*;

class UserMainCode
{
    public int isPalinNum(int input1)
    {
        int n = input1;
        int rev = 0;

        while(n > 0)
        {
            rev = rev * 10 + (n % 10);
            n /= 10;
        }

        if(rev == input1)
            return 2;
        else
            return 1;
    }
}
