import java.io.*;
import java.util.*;

class UserMainCode
{
    public int totalHillWeight(int input1, int input2, int input3)
    {
        int total = 0;
        int weight = input2;

        for(int i = 1; i <= input1; i++)
        {
            total += i * weight;
            weight += input3;
        }

        return total;
    }
}
