import java.io.*;
import java.util.*;

class UserMainCode {

    public int sumOfSumsOfDigits(int input1) {
        String str = String.valueOf(input1);
        int totalSum = 0;

        for (int i = 0; i < str.length(); i++) {
            int digit = str.charAt(i) - '0';
            totalSum += digit * (i + 1);
        }

        return totalSum;
    }
}
